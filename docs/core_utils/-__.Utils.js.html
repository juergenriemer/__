	 * Uppercases the first character of a word
	 * </pre>

	, camelCase : function( s ) {

		s.split( " " ).forEach( function( s, ix ) {

		var sMonth = lsMonths[ nMonth ];
		return {
			  nDay : nDay
			, sDay : sDay
			, sDayShort : sDay.substring( 0, 3 )
			, nMonth : nMonth
			, sMonth : sMonth
			, sMonthShort : sMonth.substring( 0, 3 )
		}
	}
};
